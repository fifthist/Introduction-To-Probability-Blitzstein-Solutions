\begin{flalign}
    \mathbb{E}[X] & = \int_{0}^{\infty} x f(x) dx \\
    & = \int_{0}^{\infty} x^2 e^{-\frac{x^2}{2}} dx \\
    & = \frac{1}{2} \ri x^2 e^{-\frac{x^2}{2}} dx \\
    & = \frac{1}{2}
\end{flalign}


\begin{flalign}
    \mathbb{E}[X^2] & = \int_{0}^{\infty} x^2 f(x) dx \\
    & = \int_{0}^{\infty} x^3 e^{\frac{x^2}{2}} dx \\
    & = \int_{0}^{\infty} 2 u e^u du = 2 \\
\end{flalign}
