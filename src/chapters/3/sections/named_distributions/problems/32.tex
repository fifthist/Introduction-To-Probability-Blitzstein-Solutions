\begin{enumerate}[label=(\alph*)]
\item The problem fits the story of Hypergeometric distributions.

$$P(X = k) = \frac{\binom{s}{k}\binom{100-s}{10-k}}{\binom{100}{10}}$$
for $0 \leq k \leq s$.

\item 
\begin{verbatim}
> x = 75

\end{verbatim}
\end{enumerate}
